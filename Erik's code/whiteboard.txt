void addNewStudent(Student arr[], int count)			//Add new student menu
{
	string lName, fName;
	int id;
	cout << "Student last name: " << endl;
	cin >> lName;
	cout << "Student first name: " << endl;
	cin >> fName;
	cout << "Student ID: " << endl;
	cin >> id;
	Student newStu(id, lName, fName);
	arr[count] = newStu;
	cout << "\nAdd successful." << endl;
}
void listAllStudents(Student arr[], int count)				//List all students
{
	for (int i = 0; i < count; i++)
	{
		Student a = arr[i];
		cout << "Id: " << a.get_id() << " " << "Name: " << a.get_lastName() << ", " << a.get_firstName() << endl;
	}
	cout << endl;
}
